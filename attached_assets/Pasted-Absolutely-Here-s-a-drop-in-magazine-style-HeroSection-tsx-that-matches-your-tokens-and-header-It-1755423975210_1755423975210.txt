Absolutely. Here’s a drop-in, magazine-style HeroSection.tsx that matches your tokens and header. It includes the glass overlay, alternative headline treatment, mobile-first layout, and accessible structure. Works in plain React; I’ve also added a tiny Next.js variant comment if you’re using next/image.

// client/src/components/HeroSection.tsx
"use client"; // (safe to keep even in CRA/Vite; ignored)

import React from "react";
import { Shield } from "lucide-react";

type HeroSectionProps = {
  title?: string;
  tagline?: string;
  imageSrc: string;        // hero image URL
  imageAlt?: string;
  coverlines?: string[];   // bullet items in the glass box
  className?: string;
};

export const HeroSection: React.FC<HeroSectionProps> = ({
  title = "IQSF",
  tagline = "Defining Safety, Advancing Equality",
  imageSrc,
  imageAlt = "",
  coverlines = [
    "Global Safety Index Analysis",
    "Interactive Map Exploration",
  ],
  className = "",
}) => {
  return (
    <section
      id="hero"
      aria-label="Feature cover"
      className={`relative w-full editorial-spacing ${className}`}
    >
      {/* Masthead lockup (optional if Header already shows it) */}
      <div className="editorial-container mb-6">
        <div className="flex items-center gap-4">
          <div className="w-10 h-10 bg-black flex items-center justify-center editorial-shadow">
            <Shield className="text-white h-5 w-5" aria-hidden="true" />
          </div>
          <div>
            <h1 className="font-display text-4xl sm:text-5xl tracking-wider text-black font-light">
              {title}
            </h1>
            <p className="caption text-black/70 tracking-[0.2em] mt-1">
              {tagline.toUpperCase()}
            </p>
          </div>
        </div>
      </div>

      {/* Hero image with overlay */}
      <div className="editorial-container">
        <div className="relative editorial-shadow editorial-overlay rounded-sm overflow-hidden">
          {/* Plain <img>. If using Next.js, swap to <Image fill priority ... /> */}
          <img
            src={imageSrc}
            alt={imageAlt}
            className="w-full h-[58vh] sm:h-[70vh] object-cover"
            draggable={false}
          />

          {/* Subtle top-left magazine title (alternate treatment) */}
          <div className="absolute top-6 left-6 z-10">
            <span className="font-display text-white/90 text-5xl sm:text-7xl tracking-tight drop-shadow-lg">
              {title}
            </span>
            <div className="mt-2">
              <span className="caption text-white/80 tracking-[0.35em]">
                {tagline.toUpperCase()}
              </span>
            </div>
          </div>

          {/* Glass coverline box */}
          <div
            className="absolute bottom-8 left-6 right-6 sm:left-8 sm:right-auto sm:max-w-md
                       backdrop-blur-sm bg-black/40 text-white p-6 editorial-shadow rounded-sm"
          >
            <p className="caption mb-3 tracking-widest">INSIDE THIS ISSUE</p>
            <ul className="space-y-1 font-serif text-base leading-relaxed">
              {coverlines.map((line, i) => (
                <li key={i}>• {line}</li>
              ))}
            </ul>
          </div>

          {/* Decorative gradient line at bottom for luxury vibe */}
          <div className="absolute inset-x-0 bottom-0 h-1 pride-gradient-editorial opacity-70" />
        </div>
      </div>
    </section>
  );
};

export default HeroSection;

Notes
	•	Glass overlay uses backdrop-blur-sm bg-black/40 so the cover image shines without losing readability.
	•	Uses your caption / font-display / serif system + tokens you defined.
	•	Responsive: hero is ~70vh on larger screens, ~58vh on mobile; tweak if you want a taller cover.
	•	If you’re in Next.js, swap the <img> with:

// import Image from "next/image";
<div className="relative w-full h-[58vh] sm:h-[70vh]">
  <Image src={imageSrc} alt={imageAlt} fill priority className="object-cover" />
</div>
